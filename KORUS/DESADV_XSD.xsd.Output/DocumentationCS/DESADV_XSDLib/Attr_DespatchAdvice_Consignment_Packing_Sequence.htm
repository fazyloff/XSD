Packing_Sequence
  Property Description  
    Property Name Packing_Sequence  
    Property Type  
    Parent Class DESADV_XSDLib.DespatchAdvice_Consignment  
    XML Type Element  
    Accessors Read Only  
    Description A collection of Packing_Sequences  
    Remarks This collection may contain 0 to Many Packing_Sequences.


  
    Example 
DESADV_XSDLib.Packing_Sequence oItem;
DESADV_XSDLib.DespatchAdvice_Consignment oParent = ...;

// The collection object will already exist, but we may want to add items to it

// This method creates a new object adds it to the collection and 
// returns the new object
oItem = oParent.Packing_Sequence.Add();

// This way you can add an existing object to the collection
oItem = new DESADV_XSDLib.Packing_Sequence();
oParent.Packing_Sequence.Add(oItem);

//-------------------------------------------------------------
// To iterate through the collection

foreach(DESADV_XSDLib.Packing_Sequence itm in oParent.Packing_Sequence)
{
  System.Diagnostics.Debug.WriteLine(itm.[Some Property ....]);
}

// using an index
for(int i=0; i<oParent.Packing_Sequence.Count; i++)
{
  oItem = oParent.Packing_Sequence[i];
  System.Diagnostics.Debug.WriteLine(oItem.[Some Property ....]);
}
